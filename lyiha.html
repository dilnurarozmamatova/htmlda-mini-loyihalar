import sqlite3 from telegram import Update, ReplyKeyboardMarkup from telegram.ext import ApplicationBuilder, CommandHandler, MessageHandler, filters, ContextTypes TOKEN = "SENING_TOKENING" # --- DATABASE --- conn = sqlite3.connect("yordam.db") cursor = conn.cursor() cursor.execute(""" CREATE TABLE IF NOT EXISTS posts ( id INTEGER PRIMARY KEY AUTOINCREMENT, user_id INTEGER, text TEXT ) """) conn.commit() # --- MENYU --- menu = ReplyKeyboardMarkup( [["🆘 Yordam so‘rash", "🤝 Yordam berish"], ["📋 E’lonlarni ko‘rish"]], resize_keyboard=True ) # --- START --- async def start(update: Update, context: ContextTypes.DEFAULT_TYPE): await update.message.reply_text( "Salom! 👋\nMahalla Yordam botiga xush kelibsiz!", reply_markup=menu ) # --- XABARLAR --- async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE): text = update.message.text user_id = update.message.from_user.id if text == "🆘 Yordam so‘rash": await update.message.reply_text("Nima yordam kerak? Yozing:") context.user_data["state"] = "waiting_help" elif text == "🤝 Yordam berish": await update.message.reply_text("Qanday yordam bera olasiz? Yozing:") context.user_data["state"] = "waiting_offer" elif text == "📋 E’lonlarni ko‘rish": cursor.execute("SELECT text FROM posts ORDER BY id DESC LIMIT 5") rows = cursor.fetchall() if not rows: await update.message.reply_text("Hozircha e’lonlar yo‘q 😔") else: msg = "📢 Oxirgi e’lonlar:\n\n" for r in rows: msg += f"- {r[0]}\n" await update.message.reply_text(msg) elif context.user_data.get("state") == "waiting_help": cursor.execute("INSERT INTO posts (user_id, text) VALUES (?, ?)", (user_id, "🆘 " + text)) conn.commit() await update.message.reply_text("✅ Yordam so‘rovingiz joylandi!", reply_markup=menu) context.user_data["state"] = None elif context.user_data.get("state") == "waiting_offer": cursor.execute("INSERT INTO posts (user_id, text) VALUES (?, ?)", (user_id, "🤝 " + text)) conn.commit() await update.message.reply_text("✅ Yordam taklifingiz joylandi!", reply_markup=menu) context.user_data["state"] = None else: await update.message.reply_text("Iltimos, menyudan foydalaning 👇", reply_markup=menu) # --- BOTNI ISHGA TUSHIRISH --- app = ApplicationBuilder().token(TOKEN).build() app.add_handler(CommandHandler("start", start)) app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message)) print("Bot ishlayapti 🚀") app.run_polling()
